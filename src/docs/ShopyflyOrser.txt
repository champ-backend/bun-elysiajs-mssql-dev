// อินเทอร์เฟซสำหรับข้อมูลลูกค้าในคำสั่งซื้อ

  name: string                // ชื่อของลูกค้า
  email: string               // อีเมลของลูกค้า

  phone?: string              // หมายเลขโทรศัพท์ของลูกค้า
  billingName?: string        // ชื่อสำหรับใบแจ้งหนี้
  billingStreet?: string      // ถนนสำหรับใบแจ้งหนี้
  billingAddress1?: string    // ที่อยู่บรรทัดที่ 1 สำหรับใบแจ้งหนี้
  billingAddress2?: string    // ที่อยู่บรรทัดที่ 2 สำหรับใบแจ้งหนี้

  billingCity?: string        // เมืองสำหรับใบแจ้งหนี้
  billingZip?: string         // รหัสไปรษณีย์สำหรับใบแจ้งหนี้
  billingProvince?: string    // จังหวัดสำหรับใบแจ้งหนี้
  billingCountry?: string     // ประเทศสำหรับใบแจ้งหนี้
  billingPhone?: string       // หมายเลขโทรศัพท์สำหรับใบแจ้งหนี้
  shippingName?: string       // ชื่อสำหรับการจัดส่ง
  shippingStreet?: string     // ถนนสำหรับการจัดส่ง
  shippingAddress1?: string   // ที่อยู่บรรทัดที่ 1 สำหรับการจัดส่ง
  shippingAddress2?: string   // ที่อยู่บรรทัดที่ 2 สำหรับการจัดส่ง
  shippingCompany?: string    // ชื่อบริษัทสำหรับการจัดส่ง
  shippingCity?: string       // เมืองสำหรับการจัดส่ง
  shippingZip?: string        // รหัสไปรษณีย์สำหรับการจัดส่ง

  shippingCountry?: string    // ประเทศสำหรับการจัดส่ง
  shippingPhone?: string      // หมายเลขโทรศัพท์สำหรับการจัดส่ง

  noteAttributes?: string     // คุณสมบัติหรือข้อมูลเพิ่มเติมในหมายเหตุ

// อินเทอร์เฟซสำหรับรายการสินค้าในคำสั่งซื้อ
interface OrderLineItem {
  quantity: number            // จำนวนสินค้าในรายการ
  name: string                // ชื่อสินค้าในรายการ
  price: number               // ราคาสินค้าในรายการ
  compareAtPrice?: number     // ราคาเปรียบเทียบของสินค้า (ถ้ามี)
  sku?: string                // รหัสสินค้า (SKU)
  requiresShipping: boolean   // ระบุว่าสินค้าต้องการการจัดส่งหรือไม่
  taxable: boolean            // ระบุว่าสินค้าต้องเสียภาษีหรือไม่
  fulfillmentStatus?: string  // สถานะการจัดส่งของสินค้า
  discount?: number           // จำนวนส่วนลดของสินค้า
}

  discountCode?: string       // รหัสส่วนลด
  discountAmount?: number     // จำนวนเงินส่วนลด
  shippingMethod?: string     // วิธีการจัดส่ง
  createdAt: Date             // วันที่สร้างคำสั่งซื้อ
  cancelledAt?: Date          // วันที่ยกเลิกคำสั่งซื้อ
  paymentMethod?: string      // วิธีการชำระเงิน
  paymentReference?: string   // อ้างอิงการชำระเงิน
  refundedAmount?: number     // จำนวนเงินที่คืน
  outstandingBalance?: number // เงินคงค้าง
  employee?: string           // ชื่อพนักงานที่เกี่ยวข้อง
  location?: string           // สถานที่ที่เกี่ยวข้อง
  deviceId?: string           // รหัสอุปกรณ์
  id: string                  // รหัสคำสั่งซื้อ
  tags?: string               // แท็กสำหรับคำสั่งซื้อ
  riskLevel?: string          // ระดับความเสี่ยง
  source?: string             // แหล่งที่มาของคำสั่งซื้อ
  paymentId?: string          // รหัสการชำระเงิน
  paymentTermsName?: string   // ชื่อเงื่อนไขการชำระเงิน
  nextPaymentDueAt?: Date     // วันที่ชำระเงินถัดไป
  paymentReferences?: string  // อ้างอิงการชำระเงินทั้งหมด
  receiptNumber?: string      // หมายเลขใบเสร็จ
  duties?: number             // ภาษีศุลกากร
}

// อินเทอร์เฟซสำหรับข้อมูลกำหนดเองในคำสั่งซื้อ
interface CustomOrderFields {
  vendor?: string                 // ชื่อผู้ขาย
  category?: string               // หมวดหมู่ของคำสั่งซื้อ
  sorApasNumber?: string          // หมายเลข SOR / Apas
  codesales?: string              // รหัสการขาย
  receiptInvoice?: string         // ใบเสร็จ/ใบกำกับภาษี
  plant?: string                  // ชื่อโรงงาน
  addressLine1?: string           // ที่อยู่บรรทัดที่ 1 (อ้างอิงจากคอลัมน์ AA: Building Address1)
  subdistrict?: string            // เขตย่อย
  city?: string                   // เมือง
  stateProvinceRegion?: string    // รัฐ/จังหวัด/ภูมิภาค
  zip?: string                    // รหัสไปรษณีย์ (อ้างอิงจากคอลัมน์ AE: Billing Zip)
  paymentReferencesRef?: string   // อ้างอิงการชำระเงิน (อ้างอิงจากคอลัมน์ AW: Payment Reference)
  grandTotal?: number             // ยอดรวมทั้งหมด (อ้างอิงจากคอลัมน์ I: Subtotal)
  normalSrp?: number              // ราคาปกติ (SRP) (อ้างอิงจากคอลัมน์ S: Lineitem price)
  discount?: number               // จำนวนส่วนลด (อ้างอิงจากคอลัมน์ N: Discount Amount)
  srpAfterPromotion?: number      // ราคาหลังโปรโมชัน (SRP) (อ้างอิงจากคอลัมน์ L: Total)
  exvat?: number                  // ยอดรวมก่อนภาษี
  grandTotalFinal?: number        // ยอดรวมทั้งหมดสุดท้าย (GRAND TOTAL)
}

// อินเทอร์เฟซหลักสำหรับคำสั่งซื้อทั้งหมด
interface Order {
  customer: OrderCustomerInfo     // ข้อมูลลูกค้า
  lineItems: OrderLineItem[]      // รายการสินค้าในคำสั่งซื้อ (อาร์เรย์)
  order: OrderInfo                // ข้อมูลคำสั่งซื้อ
  taxes: OrderTaxInfo             // ข้อมูลภาษี
  custom: CustomOrderFields       // ข้อมูลกำหนดเอง
}
